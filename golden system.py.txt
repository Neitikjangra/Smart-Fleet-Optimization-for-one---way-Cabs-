# *Enhanced Data Cleaning Functions*
def clean_numeric_string(value):
    """Handle special encodings and convert to numeric"""
    if pd.isna(value):
        return np.nan
    if isinstance(value, str):
        value = re.sub(r'^\+[A-Z0-9]+\-', '', value)
        value = re.sub(r'[^\d\.\-]', '', value)
        if value == '':
            return np.nan
    return value

def safe_convert_to_int(series):
    """Safely convert to integer handling NA values"""
    return pd.to_numeric(series, errors='coerce').astype('Int32')

# *Comprehensive Data Loader*
def load_and_clean_data(filepath):
    """Handles all data issues with robust type conversion"""
    # Load data as strings first
    df = pd.read_csv(
        filepath,
        dtype='object',
        low_memory=False,
        encoding='utf-8',
        on_bad_lines='warn'
    )

    # Fix AF8 column names
    df.columns = [col.replace('+AF8-', '_') for col in df.columns]

    # Clean and convert numeric columns
    numeric_cols = {
        'vendor_id': 'int8',
        'pickup_loc': 'int32',
        'drop_loc': 'int32',
        'driver_tip': 'float32',
        'mta_tax': 'float32',
        'distance': 'float32',
        'num_passengers': 'int8',
        'toll_amount': 'float32',
        'payment_method': 'int8',
        'rate_code': 'int8',
        'extra_charges': 'float32',
        'improvement_charge': 'float32',
        'total_amount': 'float32'
    }

    for col, dtype in numeric_cols.items():
        if col in df.columns:
            df[col] = df[col].apply(clean_numeric_string)
            if 'int' in dtype:
                df[col] = safe_convert_to_int(df[col])
            else:
                df[col] = pd.to_numeric(df[col], errors='coerce').astype(dtype)

    # Convert datetime columns
    datetime_cols = [col for col in df.columns if 'time' in col.lower()]
    for col in datetime_cols:
        df[col] = pd.to_datetime(df[col], errors='coerce')

    return df, datetime_cols

# *Step 2: Enhanced Golden Points Analysis*
def find_golden_points(df, min_trips=30):
    """Find golden points with state-wise distribution"""
    # Create trip matrix
    trip_matrix = df.groupby(['pickup_loc', 'drop_loc']).size().reset_index(name='count')

    # Find asymmetric demand
    merged = trip_matrix.merge(
        trip_matrix,
        left_on=['pickup_loc', 'drop_loc'],
        right_on=['drop_loc', 'pickup_loc'],
        how='left',
        suffixes=('', '_return')
    )

    # Filter for golden points
    golden_points = merged[
        (merged['count'] >= min_trips) &
        ((merged['count_return'].isna()) | (merged['count'] > 2 * merged['count_return']))
    ].sort_values('count', ascending=False)

    return golden_points.set_index('drop_loc')['count'].rename('one_way_trips')
